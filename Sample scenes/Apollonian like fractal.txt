  Helper Functions:

float F( vec3 p0 ){
   vec4 p = vec4(p0, 1.);
   for(int i = 0; i < 10; i++){
      p.xyz = mod(p.xyz-1.,2.)-1.;
      p*=1.7/dot(p.xyz,p.xyz);
   }
   return length(p.xz/p.w)*0.25;
}
